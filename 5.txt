#Найти квадрат данного натурального числа, если квадрат любого натурального числа n равен сумме n первых нечетных чисел. Составить таблицу квадратов натуральных чисел от 1 до 10000.

def square(n):
    count = 0
    for i in range(1, n * 2, 2):
        count += i
        if count ==  n ** 2:
            return count

for i in range(1, 10001):
    print(square(i))
