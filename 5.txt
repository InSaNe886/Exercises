#Найти квадрат данного натурального числа, если квадрат любого натурального числа n равен сумме n первых нечетных чисел. Составить таблицу квадратов натуральных чисел от 1 до 10000.

def square(n):
    count = 0
    for i in range(1, n * 2, 2):
        count += i
        if not None:
            continue
    return count

for i in range(1, 10001):
    print(square(i))


#Куб любого натурального числа n равен сумме n нечетных чисел, следующих по порядку за числами, сумма которых составила куб числа n−1. С помощью этого факта составьте таблицу кубов натуральных чисел от 1 до 100.

def qubeTabl(n):
    posl_nechetnoe = 1
    for i in range(1, n + 1):
        summa_cub = 0
        for j in range(i):
            summa_cub += posl_nechetnoe
            posl_nechetnoe += 2
        print(str(i) + " " + str(summa_cub))


print(qubeTabl(100))
